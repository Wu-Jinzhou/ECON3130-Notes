\section{Proving the Central Limit Theorem}
\textbf{Date:} \underline{Oct 22, 2025}

\begin{itemize}
    \item The Central Limit Theorem (CLT) states that for a random sample $X_1, X_2, \ldots, X_n$ from a distribution with finite mean $\mu$ and variance $\sigma^2$, the standardized sample mean
    \[
        W = \frac{\overline{X} - \mu}{\sigma / \sqrt{n}}
    \]
    converges in distribution to the standard normal:
    \[
        W \xrightarrow{d} N(0, 1)
    \]
    as $n \to \infty$.
    \item This means that as the sample size $n$ increases, the distribution of the sample mean $\overline{X}$ approaches a normal distribution with mean $\mu$ and variance $\frac{\sigma^2}{n}$:
    \[
        \overline{X} \sim N\left(\mu, \frac{\sigma^2}{n}\right)
    \]
    \item The CLT is fundamental in statistics because it justifies the use of normal approximations for inference, even when the underlying distribution is not normal.
    \item Key conditions: finite mean and variance, and sufficiently large $n$ (typically $n \geq 30$).
\end{itemize}

\subsection{The mgf Method}

\begin{itemize}
    \item If the sequence of mgfs $M_n(t)$ of random variables $X_n$ converges to an mgf $M(t)$ for $t$ in an open interval around $0$, then the distributions of $X_n$ converge to the distribution with mgf $M(t)$.
    \item In the context of the Central Limit Theorem, we show that the mgf of the standardized sum converges to the mgf of the standard normal distribution.
    \item This establishes that the limiting distribution is normal.
\end{itemize}

\subsection{Proof}

\textbf{Big Picture}
\begin{itemize}
    \item Let $W = \frac{\overline{X} - \mu}{\sigma / \sqrt{n}}$.
    \item We want to show that $W \xrightarrow{d} N(0, 1)$.
    \item The mgf of the standard normal is $\exp(t^2/2)$.
    \item Steps:
    \begin{enumerate}
        \item Derive the mgf of $W$.
        \item Take the limit of that mgf as $n \to \infty$.
        \item Show thatâ€™s the mgf of the standard normal.
    \end{enumerate}
\end{itemize}

Let $W = \frac{\overline{X} - \mu}{\sigma / \sqrt{n}} = \frac{\sqrt{n}(\overline{X} - \mu)}{\sigma} = \frac{\sum_{i=1}^n X_i - n\mu}{\sqrt{n}\sigma}$.

The moment generating function (mgf) of $W$ is:
\[
M_W(t) = E[\exp(tW)] = E\left[\exp\left(\frac{t}{\sqrt{n}\sigma} \left(\sum_{i=1}^n X_i - n\mu\right)\right)\right]
\]
\[
= E\left[\prod_{i=1}^n \exp\left(\frac{t}{\sqrt{n}\sigma}(X_i - \mu)\right)\right]
\]
Since $X_i$ are independent,
\[
= \prod_{i=1}^n E\left[\exp\left(\frac{t}{\sqrt{n}\sigma}(X_i - \mu)\right)\right]
\]
Let $Y_i = \frac{X_i - \mu}{\sigma}$ (standardized), so
\[
= \left(E\left[\exp\left(\frac{t}{\sqrt{n}} Y_i\right)\right]\right)^n
\]
Let $m(t)$ be the mgf of $Y_i$:
\[
m(t) = E[\exp(t Y_i)]
\]
So,
\[
M_W(t) = \left(m\left(\frac{t}{\sqrt{n}}\right)\right)^n
\]

Now, expand $m(t)$ using Taylor's theorem around $t=0$:
\[
m(t) = 1 + m'(0)t + \frac{m''(t_1)t^2}{2}
\]
where $t_1$ is between $0$ and $t$.

Since $E[Y_i] = 0$ and $E[Y_i^2] = 1$, we have $m(0) = 1$, $m'(0) = 0$, $m''(0) = 1$.

So,
\[
m(t) = 1 + \frac{m''(t_1)t^2}{2}
\]
\[
= 1 + \frac{t^2}{2} + \frac{(m''(t_1) - 1)t^2}{2}
\]

Plug in $t/\sqrt{n}$:
\[
m\left(\frac{t}{\sqrt{n}}\right) = 1 + \frac{t^2}{2n} + \frac{(m''(t_1) - 1)t^2}{2n}
\]
So,
\[
M_W(t) = \left[1 + \frac{t^2}{2n} + \frac{(m''(t_1) - 1)t^2}{2n}\right]^n
\]

As $n \to \infty$, $t_1 \to 0$ and $m''(t_1) \to m''(0) = 1$:
\[
\lim_{n \to \infty} M_W(t) = \lim_{n \to \infty} \left[1 + \frac{t^2}{2n}\right]^n
\]
Recall $\lim_{n \to \infty} (1 + x/n)^n = \exp(x)$:
\[
\lim_{n \to \infty} M_W(t) = \exp\left(\frac{t^2}{2}\right)
\]

This is the mgf of the standard normal distribution. Therefore, by the mgf method, $W \xrightarrow{d} N(0, 1)$ as $n \to \infty$.
